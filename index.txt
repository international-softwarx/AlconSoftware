    <script>
        // üî• CONFIGURACI√ìN DUAL
        const BACKENDS = {
            local: 'http://localhost:8080',
            production: 'https://tu-app.onrender.com'
        };
        
        let currentMode = 'local'; // 'local' o 'production'
        let BACKEND_URL = BACKENDS.local;
        
        const PASSWORD = 'admin123';
        
        let socket = null;
        let isLoggedIn = false;
        let overlayVisible = true;
        let lastScreenTime = 0;

        // üî• Selector de modo
        function selectMode(mode) {
            currentMode = mode;
            BACKEND_URL = BACKENDS[mode];
            
            // Actualizar botones
            document.querySelectorAll('.mode-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');
            
            console.log(`Modo seleccionado: ${mode} (${BACKEND_URL})`);
        }

        // Login
        function login() {
            const password = document.getElementById('passwordInput').value;
            if (password === PASSWORD) {
                isLoggedIn = true;
                document.getElementById('loginScreen').style.display = 'none';
                document.getElementById('dashboard').style.display = 'block';
                
                // üî• Actualizar badge
                const badge = document.getElementById('modeBadge');
                badge.textContent = currentMode === 'local' ? 'üè† LOCAL' : '‚òÅÔ∏è PRODUCCI√ìN';
                badge.className = currentMode === 'production' ? 'mode-badge production' : 'mode-badge';
                
                initDashboard();
            } else {
                alert('‚ùå Invalid access code');
            }
        }

        function logout() {
            isLoggedIn = false;
            document.getElementById('loginScreen').style.display = 'flex';
            document.getElementById('dashboard').style.display = 'none';
            if (socket) socket.disconnect();
        }

        // Inicializar dashboard
        function initDashboard() {
            connectWebSocket();
            checkServerStatus();
            setInterval(checkServerStatus, 30000);
        }

        // WebSocket
        function connectWebSocket() {
            socket = io(BACKEND_URL, {
                transports: ['websocket', 'polling']
            });

            socket.on('connect', () => {
                console.log(`‚úì WebSocket conectado a ${BACKEND_URL}`);
                socket.emit('client_type', { type: 'web' });
                updateStatus(true, `Connected (${currentMode})`);
            });

            socket.on('disconnect', () => {
                console.log('‚úó WebSocket desconectado');
                updateStatus(false, 'Disconnected');
            });

            socket.on('screen_update', (data) => {
                displayScreenshot(data);
            });

            socket.on('status_update', (data) => {
                document.getElementById('clientCount').textContent = data.windowsConnected || 0;
                if (data.latestScreenshot) {
                    displayScreenshot(data.latestScreenshot);
                }
            });
        }

        // Mostrar screenshot
        function displayScreenshot(data) {
            const img = document.getElementById('screenDisplay');
            const noFeed = document.getElementById('noFeed');
            const screenInfo = document.getElementById('screenInfo');
            
            img.src = 'data:image/jpeg;base64,' + data.image;
            img.style.display = 'block';
            noFeed.style.display = 'none';
            screenInfo.style.display = 'block';
            
            const now = Date.now();
            const latency = now - (data.timestamp * 1000);
            document.getElementById('latency').textContent = Math.round(latency);
            
            const time = new Date().toLocaleTimeString();
            document.getElementById('lastUpdate').textContent = time;
            
            lastScreenTime = now;
        }

        // Estado del servidor
        async function checkServerStatus() {
            try {
                const response = await fetch(`${BACKEND_URL}/ping`);
                const data = await response.json();
                
                if (!socket || !socket.connected) {
                    updateStatus(true, `Server Online (${data.windowsClients} clients)`);
                }
            } catch (error) {
                if (!socket || !socket.connected) {
                    updateStatus(false, 'Server Offline');
                }
            }
        }

        function updateStatus(online, text) {
            const dot = document.getElementById('statusDot');
            const statusText = document.getElementById('statusText');
            
            dot.className = online ? 'status-dot connected' : 'status-dot';
            statusText.textContent = text;
        }

        // Controles
        async function sendUpdate() {
            const text = document.getElementById('overlayText').value || '‚óè';
            try {
                const response = await fetch(`${BACKEND_URL}/update`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ text })
                });
                const data = await response.json();
                console.log('‚úì Sent:', text);
            } catch (error) {
                alert('‚ùå Error sending update');
            }
        }

        function sendQuick(value) {
            document.getElementById('overlayText').value = value;
            sendUpdate();
        }

        async function toggleOverlay() {
            overlayVisible = !overlayVisible;
            const toggle = document.getElementById('overlayToggle');
            toggle.classList.toggle('active');
            
            try {
                await fetch(`${BACKEND_URL}/toggle-overlay`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ visible: overlayVisible })
                });
            } catch (error) {
                console.error('Error toggling overlay');
            }
        }

        function requestScreenshot() {
            if (socket && socket.connected) {
                socket.emit('request_screenshot', {});
            }
        }

        async function wakeUpServer() {
            updateStatus(false, 'Waking up server...');
            try {
                await fetch(`${BACKEND_URL}/ping`);
                setTimeout(checkServerStatus, 2000);
            } catch (error) {
                updateStatus(false, 'Failed to wake server');
            }
        }

        // Enter para enviar
        document.addEventListener('DOMContentLoaded', () => {
            document.getElementById('passwordInput')?.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') login();
            });
            
            document.getElementById('overlayText')?.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') sendUpdate();
            });
        });
    </script>
